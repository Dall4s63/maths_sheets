\textsc{Example 1}: For the following equation, fill out 
the table of values.
$$y = 3^x.$$
\begin{tabular}{|l|l|}
\hline
x & y \\
\hline
-3 & \\
\hline
-2 & \\
\hline
-1 & \\
\hline
0 & \\
\hline
1 & \\
\hline
2 & \\
\hline
3 & \\
\hline
\end{tabular}

\vspace{3mm}

\textsc{Solution}: 

\vspace{3mm}

\begin{tabular}{|Sl|Sl|}
\hline
x & y \\
\hline
-3 & $3^{-3} = \frac{1}{3^3} = \frac{1}{27} \approx 0.037$ \\
\hline
-2 & $3^{-2} = \frac{1}{3^2} = \frac{1}{9} \approx 0.111$ \\
\hline
-1 & $3^{-1} = \frac{1}{3^1} = \frac{1}{3} \approx 0.333$ \\
\hline
0 & $3^0 = 1$ \\
\hline
1 & $3^1 = 3$ \\
\hline
2 & $3^2 = 9$ \\
\hline
3 & $3^3 = 27$ \\
\hline
\end{tabular}
