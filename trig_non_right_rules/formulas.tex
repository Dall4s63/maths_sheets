\section*{Non-Right Trigonometry Rules}
\begin{figure}[h]
\centering
\begin{tikzpicture}
    \coordinate (A) at (0, 0);
    \coordinate (B) at (5, 1);
    \coordinate (C) at (3, 4);

    \draw (A) node[below left]{\Large $A$} 
           -- node[below]{\Large $c$}
          (B) node[below right]{\Large $B$} 
           -- node[above right]{\Large $a$} 
          (C) node[above]{\Large $C$} 
           -- node[above left]{\Large $b$} (A);
    \tkzMarkAngle[size=0.6](C,B,A)
    \tkzMarkAngle[size=0.6](A,C,B)
    \tkzMarkAngle[size=0.6](B,A,C)
\end{tikzpicture}
\end{figure}
\LARGE

\textsc{Sine Rule}

$$\frac{a}{\sin(A)} = \frac{b}{\sin(B)}$$
or
$$\frac{\sin(A)}{a} = \frac{\sin(B)}{b}$$

\textsc{Cosine Rule}

$$c^2 = a^2 + b^2 - 2ab\cos(C)$$

or

$$C = \cos^{-1}\left(\frac{a^2 + b^2 - c^2}{2ab}\right)$$

\textsc{Triangle Area}

$$A = \frac{1}{2}ab\sin(C)$$
